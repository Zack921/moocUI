# 快速开始

# 安装组件库

npm i mooc-ui

# 引入组件库

在 main.js 中引入组件库

// 全部引入
import 'mooc-ui/dist/css/index.css'
import MUI from "mooc-ui";
Vue.use(MUI);

// 按需引入
import 'mooc-ui/dist/css/demo.css'
import { Demo } from "mooc-ui";
Vue.use(Demo);
